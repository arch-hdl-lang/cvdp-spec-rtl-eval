//! ---
//! tags: [fibonacci, sequence, overflow]
//! ---
//!
//! 32-bit Fibonacci sequence generator with synchronous reset and one-cycle
//! overflow reporting before the sequence automatically restarts.

/// Top-level Fibonacci generator.
///
/// Input timing: `rst` is an active-high synchronous reset and Fibonacci state
/// advances on each rising edge of `clk` while reset is low.
/// Output timing: `fib_out` and `overflow_flag` are plain outputs driven by
/// internal registers, so updates become externally visible after the clock edge
/// that writes those registers.
///
/// Transition table:
/// | Current state | Input condition | Next state | Registered outputs written on edge |
/// | Running | `next_fib` fits in 32 bits | Running | `fib_out = next_fib[31:0]`, `overflow_flag = 0` |
/// | Running | `next_fib` exceeds 32 bits | Restart | `fib_out = RegB` final valid value, `overflow_flag = 0` |
/// | Restart | `rst = 0` | Running | `fib_out = 0`, `overflow_flag = 1`, RegA/RegB restart at F(0)/F(1) |
/// | Any | `rst = 1` | Running | `fib_out = 0`, `overflow_flag = 0`, RegA/RegB reset to F(0)/F(1) |
fsm fibonacci_series
  //! RegA stores F(n-2), RegB stores F(n-1), and the Restart state supplies
  //! the delayed overflow pulse while reinitializing the sequence.

  port clk: in Clock<SysDomain>;
  port rst: in Reset<Sync>;
  port fib_out: out UInt<32>;
  port overflow_flag: out Bool;

  state [Running, Restart]
  default state Running;

  reg reg_a: UInt<32> reset rst => 0;
  reg reg_b: UInt<32> reset rst => 1;
  reg fib_out_r: UInt<32> reset rst => 0;
  reg overflow_flag_r: Bool reset rst => false;

  let reg_a_wide: UInt<33> = reg_a.zext<33>();
  let reg_b_wide: UInt<33> = reg_b.zext<33>();
  let next_fib_wide: UInt<33> = reg_a_wide +% reg_b_wide;
  let overflow_now: Bool = next_fib_wide[32];
  let next_fib_low: UInt<32> = next_fib_wide[31:0];

  let fib_out = fib_out_r;
  let overflow_flag = overflow_flag_r;

  state Running
    seq on clk rising
      overflow_flag_r <= false;
      if overflow_now
        fib_out_r <= reg_b;
      else
        reg_a <= reg_b;
        reg_b <= next_fib_low;
        fib_out_r <= next_fib_low;
      end if
    end seq

    -> Restart when overflow_now;
  end state Running

  state Restart
    seq on clk rising
      reg_a <= 0;
      reg_b <= 1;
      fib_out_r <= 0;
      overflow_flag_r <= true;
    end seq

    -> Running;
  end state Restart
end fsm fibonacci_series
