//! Fixed K=3 convolutional encoder for a serial data stream. Each sampled input bit is encoded with generator polynomials 111 and 101 using the current bit and the previous two input bits.

/// Top-level K=3 convolutional encoder.
///
/// On each rising clock edge, the module emits registered encoded bits for the current input and then advances the two-bit history. The active-high asynchronous reset clears the history and encoded outputs.
module convolutional_encoder
  port clk: in Clock<SysDomain>;
  port rst: in Reset<Async, High>;
  port data_in: in Bit;
  port encoded_bit1: out pipe_reg<Bit, 1> reset rst => 0;
  port encoded_bit2: out pipe_reg<Bit, 1> reset rst => 0;

  reg default: reset rst => 0;
  reg prev_bit1: Bit;
  reg prev_bit2: Bit;

  let next_encoded_bit1: Bit = data_in ^ prev_bit1 ^ prev_bit2;
  let next_encoded_bit2: Bit = data_in ^ prev_bit2;

  default seq on clk rising;

  seq
    encoded_bit1@1 <= next_encoded_bit1;
    encoded_bit2@1 <= next_encoded_bit2;
    prev_bit2 <= prev_bit1;
    prev_bit1 <= data_in;
  end seq
end module convolutional_encoder
