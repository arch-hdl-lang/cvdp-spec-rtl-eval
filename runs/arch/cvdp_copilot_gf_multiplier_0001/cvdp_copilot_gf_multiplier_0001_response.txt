//! Combinational 4-bit GF(2^4) multiplier using the irreducible polynomial x^4 + x + 1.
//! The datapath follows the prompt algorithm by conditionally XORing shifted multiplicands for each bit of B.

/// Top-level 4-bit Galois-field multiplier for GF(2^4).
///
/// Produces a same-cycle reduced product of A and B modulo x^4 + x + 1.
module gf_multiplier
  port A: in UInt<4>;
  port B: in UInt<4>;
  port result: out UInt<4>;

  let m0: UInt<4> = A;

  let add0: UInt<4> = B[0] ? m0 : 4'b0000;
  let sh0: UInt<4> = {m0[2:0], 1'b0};
  let m1: UInt<4> = m0[3] ? (sh0 ^ 4'b0011) : sh0;

  let add1: UInt<4> = B[1] ? m1 : 4'b0000;
  let sh1: UInt<4> = {m1[2:0], 1'b0};
  let m2: UInt<4> = m1[3] ? (sh1 ^ 4'b0011) : sh1;

  let add2: UInt<4> = B[2] ? m2 : 4'b0000;
  let sh2: UInt<4> = {m2[2:0], 1'b0};
  let m3: UInt<4> = m2[3] ? (sh2 ^ 4'b0011) : sh2;

  let add3: UInt<4> = B[3] ? m3 : 4'b0000;
  let product: UInt<4> = add0 ^ add1 ^ add2 ^ add3;

  let result = product;
end module gf_multiplier
