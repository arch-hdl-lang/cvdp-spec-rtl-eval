//! ---
//! tags: [perceptron, microcode, training, gates]
//! ---
//!
//! Microcoded perceptron trainer for the CVDP perceptron_gates task. The module sequences initialization, response calculation,
//! target selection, update, convergence check, and iteration advance on the rising clock edge.

/// Sequential microcoded controller for training a small signed perceptron on gate targets.
///
/// The controller exposes registered state for weights, bias, microcode address, selected input index,
/// thresholded response, and convergence status.
module perceptron_gates
  port clk: in Clock<SysDomain>;
  port rst_n: in Reset<Async, Low>;
  port x1: in SInt<4>;
  port x2: in SInt<4>;
  port learning_rate: in Bool;
  port threshold: in SInt<4>;
  port gate_select: in UInt<2>;

  port percep_w1: out pipe_reg<SInt<4>, 1> reset rst_n => 0;
  port percep_w2: out pipe_reg<SInt<4>, 1> reset rst_n => 0;
  port percep_bias: out pipe_reg<SInt<4>, 1> reset rst_n => 0;
  port present_addr: out pipe_reg<UInt<4>, 1> reset rst_n => 0;
  port stop: out pipe_reg<Bool, 1> reset rst_n => false;
  port input_index: out pipe_reg<UInt<3>, 1> reset rst_n => 0;
  port y_in: out pipe_reg<SInt<4>, 1> reset rst_n => 0;
  port y: out pipe_reg<SInt<4>, 1> reset rst_n => 0;
  port prev_percep_wt_1: out pipe_reg<SInt<4>, 1> reset rst_n => 0;
  port prev_percep_wt_2: out pipe_reg<SInt<4>, 1> reset rst_n => 0;
  port prev_percep_bias: out pipe_reg<SInt<4>, 1> reset rst_n => 0;

  reg default: reset rst_n => 0;
  reg target_hold: SInt<4>;
  reg delta_w1: SInt<4>;
  reg delta_w2: SInt<4>;
  reg delta_bias: SInt<4>;
  reg epoch_changed: Bool;

  let prod1_now: SInt<4> = x1 *% percep_w1;
  let prod2_now: SInt<4> = x2 *% percep_w2;
  let sum_products: SInt<4> = prod1_now +% prod2_now;
  let weighted_sum: SInt<4> = percep_bias +% sum_products;

  wire response_now: SInt<4>;
  comb
    if weighted_sum > threshold
      response_now = 1;
    elsif weighted_sum == threshold
      response_now = 0;
    else
      response_now = -1;
    end if
  end comb

  wire selected_target: SInt<4>;
  comb
    selected_target = -1;
    if gate_select == 0
      if input_index == 0
        selected_target = 1;
      else
        selected_target = -1;
      end if
    elsif gate_select == 1
      if input_index == 3
        selected_target = -1;
      else
        selected_target = 1;
      end if
    elsif gate_select == 2
      if input_index == 3
        selected_target = -1;
      else
        selected_target = 1;
      end if
    else
      if input_index == 0
        selected_target = 1;
      else
        selected_target = -1;
      end if
    end if
  end comb

  let need_update: Bool = learning_rate and (y != target_hold);

  wire wt1_update_now: SInt<4>;
  wire wt2_update_now: SInt<4>;
  wire bias_update_now: SInt<4>;
  comb
    if need_update
      wt1_update_now = x1 *% target_hold;
      wt2_update_now = x2 *% target_hold;
      bias_update_now = target_hold;
    else
      wt1_update_now = 0;
      wt2_update_now = 0;
      bias_update_now = 0;
    end if
  end comb

  let update_is_nonzero: Bool = (delta_w1 != 0) or (delta_w2 != 0) or (delta_bias != 0);
  let epoch_done: Bool = input_index == 3;
  let epoch_converged: Bool = epoch_done and (not epoch_changed) and (not update_is_nonzero);

  default seq on clk rising;

  seq
    if present_addr == 0
      percep_w1 <= 0;
      percep_w2 <= 0;
      percep_bias <= 0;
      y_in <= 0;
      y <= 0;
      stop <= false;
      input_index <= 0;
      prev_percep_wt_1 <= 0;
      prev_percep_wt_2 <= 0;
      prev_percep_bias <= 0;
      target_hold <= 0;
      delta_w1 <= 0;
      delta_w2 <= 0;
      delta_bias <= 0;
      epoch_changed <= false;
      present_addr <= 1;
    elsif present_addr == 1
      y_in <= weighted_sum;
      y <= response_now;
      present_addr <= 2;
    elsif present_addr == 2
      target_hold <= selected_target;
      present_addr <= 3;
    elsif present_addr == 3
      delta_w1 <= wt1_update_now;
      delta_w2 <= wt2_update_now;
      delta_bias <= bias_update_now;
      percep_w1 <= percep_w1 +% wt1_update_now;
      percep_w2 <= percep_w2 +% wt2_update_now;
      percep_bias <= percep_bias +% bias_update_now;
      if (wt1_update_now != 0) or (wt2_update_now != 0) or (bias_update_now != 0)
        epoch_changed <= true;
      else
        epoch_changed <= epoch_changed;
      end if
      present_addr <= 4;
    elsif present_addr == 4
      if epoch_converged
        stop <= true;
      else
        stop <= stop;
      end if
      present_addr <= 5;
    else
      prev_percep_wt_1 <= delta_w1;
      prev_percep_wt_2 <= delta_w2;
      prev_percep_bias <= delta_bias;
      if stop
        present_addr <= 5;
        input_index <= input_index;
        epoch_changed <= epoch_changed;
      else
        present_addr <= 1;
        if input_index == 3
          input_index <= 0;
          epoch_changed <= false;
        else
          input_index <= input_index +% 1;
        end if
      end if
    end if
  end seq
end module perceptron_gates
