//! Digital stopwatch with an internal one-second pulse divider. The stopwatch
//! preserves partial-second divider progress while paused and holds at exactly
//! one hour until asynchronous reset.

/// Top-level digital stopwatch implementation.
///
/// Timing/type notes:
/// - `clk` is sampled on its rising edge.
/// - `reset` is asynchronous active-high and clears divider, seconds, minutes, and hour.
/// - `seconds`, `minutes`, and `hour` are combinational views of internal registers.
///
/// Transition/operation table:
/// | Current condition | Input condition | Next condition | Externally visible outputs |
/// | reset asserted | any | cleared | seconds=0, minutes=0, hour=0 immediately after reset |
/// | not at one hour | start_stop=0 | hold | seconds/minutes/hour hold; divider partial count holds |
/// | running, divider not full | start_stop=1 | same elapsed time, divider+1 | seconds/minutes/hour hold |
/// | running, divider full, seconds<59 | start_stop=1 | seconds+1 | seconds increments on clk edge; minutes/hour hold |
/// | running, divider full, seconds=59, minutes<59 | start_stop=1 | next minute | seconds=0, minutes increments, hour holds |
/// | running, divider full, seconds=59, minutes=59 | start_stop=1 | one-hour terminal hold | seconds=0, minutes=0, hour=1 |
/// | one-hour terminal hold | any start_stop | hold | seconds=0, minutes=0, hour=1 until reset |
module dig_stopwatch
  param CLK_FREQ: const = 50000000;
  local param DIV_WIDTH: const = $clog2(CLK_FREQ + 1);

  port clk: in Clock<SysDomain>;
  port reset: in Reset<Async, High>;
  port start_stop: in Bool;
  port seconds: out UInt<6>;
  port minutes: out UInt<6>;
  port hour: out Bool;

  reg default: reset reset => 0;
  reg div_count: UInt<DIV_WIDTH>;
  reg sec_value: UInt<6>;
  reg min_value: UInt<6>;
  reg hour_value: Bool;

  let seconds = sec_value;
  let minutes = min_value;
  let hour = hour_value;

  let div_last: Bool = (div_count == (CLK_FREQ - 1));
  let sec_last: Bool = (sec_value == 6'd59);
  let min_last: Bool = (min_value == 6'd59);

  default seq on clk rising;

  seq
    if hour_value
      div_count <= 0;
      sec_value <= 0;
      min_value <= 0;
      hour_value <= true;
    elsif start_stop
      if div_last
        div_count <= 0;
        if sec_last
          sec_value <= 0;
          if min_last
            min_value <= 0;
            hour_value <= true;
          else
            min_value <= min_value +% 1;
          end if
        else
          sec_value <= sec_value +% 1;
        end if
      else
        div_count <= div_count +% 1;
      end if
    end if
  end seq
end module dig_stopwatch
