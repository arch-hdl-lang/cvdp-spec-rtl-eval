//! Clock jitter detection by counting primary clock cycles between sampled rising edges of system_clk.
//! The design uses synchronous reset and produces a one-cycle registered jitter_detected pulse for intervals outside JITTER_THRESHOLD +/- 1.

/// Counts clk cycles between consecutive sampled rising edges of system_clk and flags interval jitter.
///
/// Timing: system_clk is sampled on the rising edge of clk. jitter_detected is a registered one-cycle pulse visible one clk after an out-of-range completed interval is observed.
module clock_jitter_detection_module
  param JITTER_THRESHOLD: const = 5;

  port clk: in Clock<SysDomain>;
  port system_clk: in Bool;
  port rst: in Reset<Sync, High>;
  port jitter_detected: out pipe_reg<Bool, 1> reset rst => false;

  reg default: reset rst => 0;
  reg edge_count: UInt<32>;
  reg edge_count_r: UInt<32>;
  reg prev_system_clk: Bool;
  reg have_edge: Bool;

  let edge_seen: Bool = system_clk and not prev_system_clk;
  let completed_interval: UInt<32> = edge_count +% 32'd1;
  let late_jitter: Bool = completed_interval > (JITTER_THRESHOLD + 1);
  let early_margin: UInt<33> = completed_interval + 32'd1;
  let early_jitter: Bool = early_margin < JITTER_THRESHOLD;
  let interval_jitter: Bool = early_jitter or late_jitter;
  let jitter_now: Bool = edge_seen and have_edge and interval_jitter;

  default seq on clk rising;

  seq
    prev_system_clk <= system_clk;
    jitter_detected <= jitter_now;

    if edge_seen
      if have_edge
        edge_count_r <= completed_interval;
      else
        edge_count_r <= 0;
        have_edge <= true;
      end if
      edge_count <= 0;
    elsif have_edge
      edge_count <= edge_count +% 32'd1;
    else
      edge_count <= 0;
    end if
  end seq
end module clock_jitter_detection_module
